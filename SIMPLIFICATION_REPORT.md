def get_accuracy_score(self, data: Data, test_answer: str) -> float:

    # –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ–π –ø–æ–≥—Ä–µ—à–Ω–æ—Å—Ç–∏

    else:
        relative_error = abs(agent_value - correct_value) / abs(correct_value)

    # –ì—Ä–∞–¥–∏–µ–Ω—Ç–Ω–∞—è –æ—Ü–µ–Ω–∫–∞

    else:
